Examples

Type alerts


    <link href="coreui-alert.min.css" rel="stylesheet"/>
    <script src="coreui-alert.min.js"></script>

    <div id="example-success"></div>
    <div id="example-info"></div>
    <div id="example-warning"></div>
    <div id="example-danger"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

            // Success
            var alertSuccess = new CoreUIAlert({
                "type": "alert",
                "view": "success",
                "title": "Success title!",
                "message": "You successfully read this important alert message."
            });
            alertSuccess.render('example-success');

            // Info
            var alertInfo = new CoreUIAlert({
                "type": "alert",
                "view": "info",
                "title": "Info title!",
                "message": "Best check yo self, you're not looking too good."
            });
            alertInfo.render('example-info');

            // Warning
            var alertWarning = new CoreUIAlert({
                "type": "alert",
                "view": "warning",
                "title": "Warning title!",
                "message": "Better check yourself, you're not looking too good."
            });
            alertWarning.render('example-warning');

            // Danger
            var alertDanger = new CoreUIAlert({
                "type": "alert",
                "view": "danger",
                "title": "Danger title!",
                "message": "Raw denim you probably haven't heard of them jean shorts Austin. Nesciunt tofu stumptown aliqua, retro synth master cleanse. Mustache cliche tempor, williamsburg carles vegan helvetica. Reprehenderit butcher retro keffiyeh dreamcatcher synth. Cosby sweater eu banh mi, qui irure terry richardson ex squid. Aliquip placeat salvia cillum iphone. Seitan aliquip quis cardigan american apparel, butcher voluptate nisi qui."
            });
            alertDanger.render('example-danger');
        });
    </script>
                            

    $alert_success = CoreUI\Alert::success("You successfully read this important alert message.", "Success title!");
    $alert_info    = CoreUI\Alert::info("Best check yo self, you're not looking too good.",       "Info title!");
    $alert_warning = CoreUI\Alert::warning("Better check yourself, you're not looking too good.", "Warning title!");
    $alert_danger  = CoreUI\Alert::danger("Raw denim you probably haven't heard of them jean...", "Danger title!");


    echo json_encoded($alert_success); // JSON output success
    echo json_encoded($alert_info);    // JSON output info
    echo json_encoded($alert_warning); // JSON output warning
    echo json_encoded($alert_danger);  // JSON output danger


Custom content


    <link href="coreui-alert.min.css" rel="stylesheet"/>
    <script src="coreui-alert.min.js"></script>

    <div id="example-custom-content"></div>

    <script>
        document.addEventListener('DOMContentLoaded', function () {

            // Custom content
            var alertCustom = new CoreUIAlert({
                "type": "alert",
                "view": "danger",
                "title": "Oh snap! You got an error!",
                "message":
                    '<p>Change this and that and try again. Duis mollis, est non commodo luctus, nisi erat porttitor ligula, eget lacinia odio sem nec elit. Cras mattis consectetur purus sit amet fermentum.</p><p><button type="button" class="btn btn-danger">Take this action</button> <button type="button" class="btn btn-default">Or do this</button> </p>'
            });
            alertCustom.render('example-custom-content');
        });
    </script>
                            

    $alert_danger = CoreUI\Alert::danger(
        '<p>Change this and that and try again. Duis mollis, est non commodo luctus, nisi erat porttitor ligula, eget lacinia odio sem nec elit. Cras mattis consectetur purus sit amet fermentum.</p><p><button type="button" class="btn btn-danger">Take this action</button> <button type="button" class="btn btn-default">Or do this</button> </p>',
        "Oh snap! You got an error!"
    );

    echo json_encoded($alert_danger); // JSON output danger

Options

Name type default description
type string The data type. The alert must be specified for correct interpretation.
view string warning Identify the appearance of the alert. Can have different values: info, success, warning, danger. But there may be other values if you have you will create your personal styles.
title string The title of the alert.
message string The text content of the message. It is allowed html.

Methods

alert.render('element-id')

Forms and inserts the generated html into the specified element. The element parameter can contain the page element identifier or the page element object. If the specified parameter is not present, the method will return html.

Browser compatibility

CoreUI Alert works in most major browsers. Some details:

  • IE: 9.
  • Microsoft Edge: 12+
  • Safari: 4+
  • Firefox: 4+
  • Chrome 14+
  • Opera: 15+
